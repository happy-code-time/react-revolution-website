import * as React from 'react';

import ReactDOM from 'react-dom';

import FullScreenListArray from './module/source/FullScreenListArray';

import './module/source/sass/rr-fullscreenlistarray.scss';

/**
 * Render it to the html while testing
 */

class App extends React.Component 
{
    constructor(props) {
        super(props);
        this.toggleList = this.toggleList.bind(this);
        this.setValue = this.setValue.bind(this);

        this.state = {
            display: false
        }
    }

    setValue(clickEvent, value){
        console.info(value);
    }

    toggleList(){
        this.setState({ display: !this.state.display });
    }

    render() {
        const { display } = this.state;

        return (
            <span>
                <button onClick={() => this.toggleList()}>Click to toggle</button>
                <FullScreenListArray
                    display={display}
                    callback={this.setValue}
                    iconClose="âœ–"
                    callbackClose={this.toggleList}
                    closeOnCallback={false}
                    closeOnDimmedClick={true}
                    closeOnEsc={true}
                    inputActive={true}
                    inputPlaceholder={'Deutschland, Polska ....'}
                    noDataText={'No data found'}
                    animation='top' // scale, top, right, bottom, left
                    data={
                        [
                            '1',
                            '2',
                            '3',
                            '4',
                            '5'
                        ]
                    }
                />
            </span>
        )
    }
}

ReactDOM.render(<App />, document.getElementById('app'));
